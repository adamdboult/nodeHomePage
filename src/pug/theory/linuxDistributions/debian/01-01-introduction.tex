
\subsection{apt-get}

package updates

to get list of packages possible updates

\begin{verbatim}
apt-get update
\end{verbatim}

to upgrade packages
\begin{verbatim}
apt-get upgrade
\end{verbatim}

debian install package

\begin{verbatim}
apt-get install <package>
\end{verbatim}

purge removes conf files too
\begin{verbatim}
apt-get remove <package>
apt-get purge <package>
\end{verbatim}
