\subsection{malloc}
malloc (memory allocation):
\begin{verbatim}
int * ptr = (int *) malloc(10 * sizeof(int));
\end{verbatim}

\subsection{calloc}
calloc (contiguous allocation):
\begin{verbatim}
int * ptr = (int *) calloc(10, sizeof(int));
\end{verbatim}

calloc initiates all to 0

\subsection{free}
can free
\begin{verbatim}
int * ptr = (int *) malloc(10 * sizeof(int));
free(ptr);
\end{verbatim}

or
\begin{verbatim}
int * ptr = (int *) calloc(10, sizeof(int));
free(ptr);
\end{verbatim}


\subsection{realloc}
\begin{verbatim}
int * ptr = (int *) malloc(10 * sizeof(int));
ptr = realloc(ptr, 20*sizeof(int));
\end{verbatim}
