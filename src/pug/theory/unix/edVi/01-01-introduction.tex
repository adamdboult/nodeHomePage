\subsection{Splitting window}

Split horizontal.
\begin{verbatim}
:split
\end{verbatim}

Split vertical
\begin{verbatim}
:vsplit
\end{verbatim}


to navigate between windows
\begin{verbatim}
ctrl-W h/j/k/l
\end{verbatim}

to resize
\begin{verbatim}
ctrl-W +-<>
\end{verbatim}

to reset sizes
\begin{verbatim}
ctrl-W =
\end{verbatim}

to swap window position
\begin{verbatim}
ctrl-W HJKL
\end{verbatim}
