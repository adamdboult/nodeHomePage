
\subsection{Linux modules}
mods are in /lib/modules/

\subsection{Linux module commands}

Show loaded modules
\begin{verbatim}
lsmod
\end{verbatim}

install mods

\begin{verbatim}
insmod
\end{verbatim}

\begin{verbatim}
rmmod
\end{verbatim}

load mod and dependencies
\begin{verbatim}
modprobe
\end{verbatim}

Get information on a module
\begin{verbatim}
modinfo
\end{verbatim}
